Bubble Sort Insertion Sort Selection Sort

function bubbleSort(array, length){

let i, fim, aux;

for (fim = length - 1; fim > 0; fim--){

for (i = 0; i < fim; i++) {

if(array[i] > array[i + 1]){

aux = array[i];

array[i] = array[i + 1];

array[i + 1] = aux;

}

}

}

}
